# ===== Core Bot Framework =====
# [webhooks] installs dependencies for running the bot with a webhook.
# [job-queue] installs dependencies for scheduled tasks like daily reminders.
python-telegram-bot[webhooks,job-queue]

# ===== Environment & Configuration =====
# For loading environment variables from a .env file.
python-dotenv

# ===== Networking =====

# [http2] is included for performance improvements.
httpx[http2]

# ===== Date & Time =====

jdatetime

# ===== Utilities =====

qrcode
